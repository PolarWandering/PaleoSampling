fig, ax = plt.subplots(figsize=(14,6))

def rename_outlier_method(x):
    if x == "vandamme": 
        return "Vandamme"
    elif x == "False":
        return "No detection"
    elif x == "True":
        return "Perfect detection"
    else: 
        raise ValueError()

df_["Sampling Strategy"] = df_.apply(lambda row : rename_outlier_method(row.ignore_outliers), axis=1) 
        
my_pal = {"Vandamme": "#e84118",
